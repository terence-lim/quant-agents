from qrafti import Panel, DATES
import pandas as pd
import json

def robust_quantile_ranking(df: pd.DataFrame, cuts: int, ascending: bool) -> pd.Series:
    """
    Ranks the data into quantiles robustly by first ranking the data to handle ties,
    then applying qcut.
    """
    # The apply method passes a DataFrame for each date. We extract the first column.
    series = df.iloc[:, 0]

    # Use 'first' method to ensure unique ranks, breaking ties arbitrarily but consistently.
    ranks = series.rank(method='first', ascending=ascending, na_option='keep')

    # Now that ranks are unique, qcut will not fail with non-monotonic bin edges.
    # We add 1 to the labels to make them 1-based (1 to 10).
    quantiles = pd.qcut(ranks, q=cuts, labels=False, duplicates='drop') + 1

    # Return as float to accommodate NaNs and align with Panel's float type.
    return quantiles.astype('float64')

# Configuration
panel_id = "_184"
cuts = 10
ascending = True

# Load the panel
data_panel = Panel().load(panel_id, **DATES)

# Apply the robust ranking function cross-sectionally (for each date)
ranked_panel = data_panel.apply(robust_quantile_ranking, cuts=cuts, ascending=ascending)

# Save the resulting panel
result_panel = ranked_panel.save()

# Print the result payload
print(json.dumps(result_panel.as_payload()))
