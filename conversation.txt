from qrafti import Panel, DATES
import pandas as pd
import numpy as np
import json

def ewma_volatility_helper(df: pd.DataFrame, decay_rate: float) -> pd.Series:
    """
    Computes the exponentially weighted moving average volatility for a single stock.
    Assumes the first column of the DataFrame contains the returns.
    """
    # Select the first column for calculation
    returns_series = df.iloc[:, 0]
    
    # Square the returns to get variance
    squared_returns = returns_series ** 2
    
    # Compute the EWMA of the squared returns (EWMA variance)
    # alpha = 1 - decay_rate
    ewma_variance = squared_returns.ewm(alpha=1 - decay_rate, adjust=False).mean()
    
    # Return the square root of the EWMA variance (volatility)
    return np.sqrt(ewma_variance)

# Load the monthly stock returns panel
returns_panel = Panel().load("RET", **DATES)

# Define the decay rate
DECAY_RATE = 0.94

# Apply the helper function to each stock's time series using trend
ewma_volatility_panel = returns_panel.trend(ewma_volatility_helper, decay_rate=DECAY_RATE)

# Save the resulting panel
result_panel = ewma_volatility_panel.save()

# Print the result payload as JSON
print(json.dumps(result_panel.as_payload()))
